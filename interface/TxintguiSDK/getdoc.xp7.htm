Using Tab-Controls is a little bit tricky ;-) I will append a piece of code to the end of this Posting... But I will explain at first...
 
1. You need to initialize Common Controls using InitCommonControls(); 2. Load the Dialog Templates into memory using FindResource(),
  LoadResource() and LockResource().
3. Set the current Tab Page using TabCtrl_GetCurSel();
4. Don´t forget to call ShowWindow() !
 
bye
  Michael
 
/*
* Little piece of code ;-)
*/
HWND          HostsGlobalTabCurrent;
HWND          HostsGlobalTabControl;
HWND          HostsGlobalTabHandle[4];
DLGTEMPLATE  *HostsGlobalTabPage[4];
 
 
/* This is the Dialog containing a TAB Control (IDT_TAB) !!! */
BOOL CALLBACK CtHostsDlgProc(HWND WinHandle, UINT Message, WPARAM WParam, LPARAM LParam)
{
  static TC_ITEM TabItem;
  static char    Buffer[256];
 
  switch(Message)
  {
      case WM_INITDIALOG:
          HostsGlobalTabControl = GetDlgItem(WinHandle, IDT_TAB);
 
          TabItem.mask    = TCIF_TEXT | TCIF_IMAGE;
          TabItem.iImage  = -1;
          TabItem.pszText = "Allgemein";
          TabCtrl_InsertItem(HostsGlobalTabControl, 0, &TabItem);
 
          TabItem.pszText = "FTP";
          TabCtrl_InsertItem(HostsGlobalTabControl, 1, &TabItem);
 
          TabItem.pszText = "TELNET";
          TabCtrl_InsertItem(HostsGlobalTabControl, 2, &TabItem);
 
          TabItem.pszText = "TN3270";
          TabCtrl_InsertItem(HostsGlobalTabControl, 3, &TabItem);
 
          HostsGlobalTabPage[0] = (DLGTEMPLATE *)
LockResource(LoadResource(GlobalInstance, FindResource(NULL,
MAKEINTRESOURCE(IDD_HOSTS_MISC), RT_DIALOG)));
          HostsGlobalTabPage[1] = (DLGTEMPLATE *)
LockResource(LoadResource(GlobalInstance, FindResource(NULL,
MAKEINTRESOURCE(IDD_HOSTS_FTP), RT_DIALOG)));
          HostsGlobalTabPage[2] = (DLGTEMPLATE *)
LockResource(LoadResource(GlobalInstance, FindResource(NULL,
MAKEINTRESOURCE(IDD_HOSTS_TELNET), RT_DIALOG)));
          HostsGlobalTabPage[3] = (DLGTEMPLATE *)
LockResource(LoadResource(GlobalInstance, FindResource(NULL,
MAKEINTRESOURCE(IDD_HOSTS_TN3270), RT_DIALOG)));
 
 
          ShowWindow(WinHandle, SW_SHOWNORMAL);
 
          CtHostsTabChange(WinHandle);
          break;
 
      /* You get this if the user want´s to change the TAB Page */       case WM_NOTIFY:
          if (((LPNMHDR) LParam)->code == TCN_SELCHANGE)
              CtHostsTabChange(WinHandle);
          break;
 
 
      default:
          return(FALSE);
 
  }
 
  return (0);
}
 
 
 
/* Change ths TAB Page... also called the first time !!! */
int CtHostsTabChange(HWND WinHandle)
{
    static int  NewPage;
 
    if (HostsGlobalTabCurrent != NULL)
      DestroyWindow(HostsGlobalTabCurrent);
 
    NewPage                  = TabCtrl_GetCurSel(HostsGlobalTabControl);
 
    if(NewPage == 0)
      HostsGlobalTabCurrent = CreateDialogIndirect(GlobalInstance,                                                     HostsGlobalTabPage[NewPage],                                                     WinHandle,
                                                    CtHostsMiscDlgProc);
 
    if(NewPage == 1)
      HostsGlobalTabCurrent = CreateDialogIndirect(GlobalInstance,                                                     HostsGlobalTabPage[NewPage],                                                     WinHandle,
                                                    CtHostsFtpDlgProc);
 
    if(NewPage == 2)
      HostsGlobalTabCurrent = CreateDialogIndirect(GlobalInstance,                                                     HostsGlobalTabPage[NewPage],                                                     WinHandle,
                                                    CtHostsTelnetDlgProc);
 
    if(NewPage
== 3)
      HostsGlobalTabCurrent = CreateDialogIndirect(GlobalInstance,
 
HostsGlobalTabPage[NewPage],
                                                    WinHandle,
                                                    CtHostsTn3270DlgProc);
 
    ShowWindow(HostsGlobalTabCurrent, SW_SHOW);
 
    return(TRUE);
}
 
 
/* Center the TAB Page within the TAB Control */
int CtHostsTabCenter(HWND WinHandle)
{
  static RECT WinRect;
  static RECT TabRect;
 
  GetWindowRect(HostsGlobalTabControl, &TabRect);
  GetWindowRect(GetParent(WinHandle), &WinRect);
 
  WinRect.left = TabRect.left - WinRect.left;
  WinRect.top  = TabRect.top  - WinRect.top -
GetSystemMetrics(SM_CYCAPTION);
 
  TabCtrl_AdjustRect(HostsGlobalTabControl, FALSE, &WinRect);
 
  SetWindowPos(WinHandle, HWND_TOP, WinRect.left, WinRect.top, 0, 0, SWP_NOSIZE);
 
  return(TRUE);
}
 
 
/* This is the dialog of a TAB Page.... */
BOOL CALLBACK CtHostsMiscDlgProc(HWND WinHandle, UINT Message, WPARAM WParam, LPARAM LParam)
{
  static RECT WinRect;
  static RECT TabRect;
  static char Temp[256];
  static LONG Index;
 
  switch(Message)
  {
      case WM_INITDIALOG:
          CtHostsTabCenter(WinHandle);  /* CENTER THE TAB PAGE ! */           break;
 
      default:
          return(FALSE);
  }
 
  return(0);
}
